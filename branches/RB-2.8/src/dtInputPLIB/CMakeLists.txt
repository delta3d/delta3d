
SET(LIB_NAME dtInputPLIB)

  ${DTCORE_LIBRARY}
)

   TARGET_LINK_LIBRARIES(${LIB_NAME} winmm)
   SET(CMAKE_SHARED_LINKER_FLAGS "/NODEFAULTLIB:libcmt;libcmtd")
