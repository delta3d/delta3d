SET(LIB_NAME     dtABC)


SET(HEADER_PATH ${DELTA3D_SOURCE_DIR}/inc/${LIB_NAME})
SET(SOURCE_PATH ${DELTA3D_SOURCE_DIR}/src/${LIB_NAME})

    ${LIB_PUBLIC_HEADERS}
    ${LIB_SOURCES}
)

											  DTUTIL_LIBRARY
											  DTCORE_LIBRARY
											  DTDAL_LIBRARY)

												 ${FLTK_LIBRARY} ${OPENGL_LIBRARY} ${MATH_LIBRARY})
