
if (RTI_INCLUDE_DIR)

SET(LIB_NAME     dtHLAGM_rti13)

SET(DTHLAGM_RTI13_LIB_BUILT ${LIB_NAME})

INCLUDE_DIRECTORIES(${RTI_INCLUDE_DIR})

file(GLOB LIB_PUBLIC_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}/*.h")

SET(LIB_SOURCES
rti13ambassador.cpp
rti13region.cpp
rti13util.cpp
)

ADD_LIBRARY(${LIB_NAME} SHARED
    ${LIB_PUBLIC_HEADERS}
    ${LIB_SOURCES}
)

LINK_WITH_VARIABLES(${LIB_NAME}
                     OSGSIM_LIBRARY
                     OPENTHREADS_LIBRARY
                   )

TARGET_LINK_LIBRARIES(${LIB_NAME}
                      ${DTUTIL_LIBRARY}
                      ${DTGAME_LIBRARY}
                      ${DTHLAGM_LIBRARY}
                      ${RTI_LIBRARIES}
                     )

IF (WIN32)
   TARGET_LINK_LIBRARIES(${LIB_NAME} wsock32)
ENDIF (WIN32)
